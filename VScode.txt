ctrl + n = new file
ctrl + shift + n = new window
ctrl + , = settings
ctrl+ o = open file
ctrl + k ctrl + o = open folder
ctrl + s = save
ctrl + shift + s = save as
ctrl + k s = save all
ctrl + f4 = close editor
ctrl + k f = close workspace
alt + f4 = close window
ctrl + z = undo
ctrl + y = redo
ctrl + x = cut
ctrl + c = copy
ctrl + v = paste
ctrl + f = find
ctrl + h = replace
ctrl + shift + f = find in files
ctrl + shift + h = find in replace
