ctrl + n = new file
ctrl + shift + n = new window
ctrl + , = settings
ctrl+ o = open file
ctrl + k ctrl + o = open folder
ctrl + s = save
ctrl + shift + s = save as
ctrl + k s = save all
ctrl + f4 = close editor
ctrl + k f = close workspace
alt + f4 = close window
ctrl + z = undo
ctrl + y = redo
ctrl + x = cut
ctrl + c = copy
ctrl + v = paste
ctrl + f = find
ctrl + h = replace
ctrl + shift + f = find in files
ctrl + shift + h = find in replace
ctrl + a = select all
shift + alt + right arrow = expand selection
shift + alt + left arrow = shrink selection
shift + alt + up arrow = copy up
shift + alt + down arrow = copy down
alt + up arrow = move line up
alt + down arrow = move line down
ctrl + d = add next occurrences
ctrl + shift + p = command palette
ctrl + shift + e = explorer
ctrl + shift + f = search
ctrl + shift + g = source control
ctrl + shift + d = run
ctrl + shift + x = extensions
ctrl + shift + m = problems
ctrl + shift + u = output
ctrl + shift + y = debug console
